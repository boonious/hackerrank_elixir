(λx.x+1)3 = (x+1)[x:=3]
          = (x[x:=3])+(1[x:=3])
          = 3 + 1
          = 4

(λx.x+1)((λy.y+2)3) = (λx.x+1)((y+2)[y:=3])
                    = (λx.x+1)( (y[y:=3]) + (2[y:=3]) )
                    = (λx.x+1)( 3 + 2 ) 
                    = (λx.x+1)(5)
                    = (x[x:=5])+(1[x:=5])
                    = 5 + 1
                    = 6
